package basic.inflearn;

import java.util.Scanner;

public class Main {

    public int[] solution(int n, int[] arr) {

        return arr;
    }

    public static void main(String[] args) {

        //Input:    6
        //          11 7 5 6 10 9
        //Output:   5 6 7 9 10 11

        Main T = new Main();
        Scanner kb = new Scanner(System.in);
